{
	//Felder
	Field x:int
	Field y:int

	PRIVATE
	{
		Field Image:TImage	
	}
	
	//Methoden
	method Draw ()
	{
		self.Image.Drawimage(self.x,self.y) //This geht auch
	}
	method New()
	{
		Basic.Print("Auto wurde erstellt")
		Carlist.Listaddlast(self)
	}
	method Delete()
	{
		Basic.Print("Auto wurde gelˆscht")
		Carlist.RemoveList(Self)
	}
}
